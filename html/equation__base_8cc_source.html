634 template <int dim>
635 double EquationBase<dim>::estimate_fiss_src
636 (std::vector<Vector<double> > &phis_this_process)
637 {
638  // first, estimate local fission source
639  double fiss_src = 0.0;
640  for (unsigned int ic=0; ic<local_cells.size(); ++ic)
641  {
642  typename DoFHandler<dim>::active_cell_iterator cell = local_cells[ic];
643  std::vector<std::vector<double> > local_phis (n_group,
644  std::vector<double> (n_q));
645  unsigned int material_id = cell->material_id ();
646  if (is_material_fissile[material_id])
647  {
648  fv->reinit (cell);
649  for (unsigned int g=0; g<n_group; ++g)
650  fv->get_function_values (phis_this_process[g],
651  local_phis[g]);
652  for (unsigned int qi=0; qi<n_q; ++qi)
653  for (unsigned int g=0; g<n_group; ++g)
654  fiss_src += (all_nusigf[material_id][g] *
655  local_phis[g][qi] *
656  fv->JxW(qi));
657  }
658  }
659  // then, we need to accumulate fission source from other processors as well
660  return Utilities::MPI::sum (fiss_src, MPI_COMM_WORLD);
661 }
